function renderTables() { const tables = getTables(); const grid = document.getElementById('tablesGrid'); grid.innerHTML = ''; for (let i = 1; i <= TABLES_COUNT; i++) { const table = tables[i] || { orders: [], total: 0 }; // ✅ Считаем ТОЛЬКО открытые заказы const openOrders = table.orders ? table.orders.filter(o => o.status !== 'closed') : []; const orderCount = openOrders.length; const isOccupied = orderCount > 0; const totalSum = openOrders.reduce((sum, o) => sum + o.total, 0); const card = document.createElement('div'); card.className = 'table-card'; card.innerHTML = ` ${isOccupied ? '🟢' : '⚪'}
Стол ${i}
${isOccupied ? `Занят (${orderCount} зак.)` : 'Свободен'} ${isOccupied ? `
${totalSum} ₽
` : ''} `; card.addEventListener('click', () => { window.location.href = `table.html?table=${i}`; }); grid.appendChild(card); } // Обновляем статистику const stats = getStats(); document.getElementById('totalOrders').textContent = stats.totalOrders || 0; document.getElementById('totalRevenue').textContent = (stats.totalRevenue || 0) + ' ₽'; let active = 0; for (let i = 1; i <= TABLES_COUNT; i++) { if (tables[i] && tables[i].orders) { const openOrders = tables[i].orders.filter(o => o.status !== 'closed'); if (openOrders.length > 0) active++; } } document.getElementById('activeTables').textContent = active; // Название заведения document.querySelector('h1').textContent = '🍽️ ' + getVenueName(); document.getElementById('venueSubtitle').textContent = `Выберите стол для работы`; }
